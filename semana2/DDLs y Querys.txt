-- 1. Listado de inscripciones con nombre de estudiante y curso
SELECT 
    e.nombre_completo AS nombre,
    c.nombre_curso AS curso,
    i.fecha_inscripcion
FROM inscripcion i
JOIN estudiante e ON i.id_estudiante = e.id_estudiante
JOIN curso c      ON i.id_curso = c.id_curso;

-- 2. Docentes con más de 5 años de experiencia
SELECT 
    c.nombre_curso AS curso, 
    d.nombre_completo AS docente, 
    d.anos_experiencia AS experiencia 
FROM curso c 
JOIN docente d ON c.id_docente = d.id_docente 
WHERE d.anos_experiencia > 5 
ORDER BY experiencia ASC;

-- 3. Promedio de calificaciones por curso
SELECT 
    c.nombre_curso AS curso, 
    ROUND(AVG(i.calificacion_final), 2) AS promedio 
FROM inscripcion i 
JOIN curso c ON i.id_curso = c.id_curso 
GROUP BY c.nombre_curso 
ORDER BY promedio DESC;

-- 4. Estudiantes matriculados en más de una materia
SELECT 
    e.nombre_completo AS estudiantes, 
    COUNT(c.id_curso) AS materias_matriculadas 
FROM inscripcion i 
JOIN estudiante e ON i.id_estudiante = e.id_estudiante 
JOIN curso c      ON i.id_curso = c.id_curso 
GROUP BY e.nombre_completo 
HAVING COUNT(*) > 1;

-- 5. Agregar columna de estado académico a la tabla estudiante
ALTER TABLE estudiante 
ADD COLUMN estado_academico VARCHAR(100) NOT NULL;

-- 6. Eliminar un docente específico por ID
DELETE FROM docente 
WHERE id_docente = 2;

-- 7. Cursos que tienen más de 2 estudiantes inscritos
SELECT 
    c.nombre_curso AS curso, 
    COUNT(i.id_estudiante) AS estudiantes 
FROM inscripcion i 
JOIN curso c      ON i.id_curso = c.id_curso 
JOIN estudiante e ON i.id_estudiante = e.id_estudiante 
GROUP BY c.nombre_curso 
HAVING COUNT(*) > 2;

-- 8. Estudiantes con promedio superior al promedio general de la institución
SELECT 
    e.id_estudiante, 
    e.nombre_completo, 
    ROUND(AVG(i.calificacion_final), 2) AS promedio_estudiante
FROM estudiante e
JOIN inscripcion i ON e.id_estudiante = i.id_estudiante
GROUP BY e.id_estudiante, e.nombre_completo
HAVING AVG(i.calificacion_final) > (
    SELECT AVG(calificacion_final) 
    FROM inscripcion
);

-- 9. Nombres de carreras con estudiantes inscritos en cursos del semestre ≥ 2 con IN
SELECT DISTINCT carrera_estudiante
FROM estudiante
WHERE id_estudiante IN (
    SELECT i.id_estudiante
    FROM inscripcion i
    JOIN curso c ON i.id_curso = c.id_curso
    WHERE c.semestre >= 2
);

-- 9.1. Nombres de carreras con estudiantes inscritos en cursos del semestre ≥ 2 con EXISTS
SELECT DISTINCT e.carrera_estudiante
FROM estudiante e
WHERE EXISTS (
    SELECT 1
    FROM inscripcion i
    JOIN curso c ON i.id_curso = c.id_curso
    WHERE i.id_estudiante = e.id_estudiante
      AND c.semestre >= 2
);

-- 10. INDICADORES
SELECT 
    e.carrera_estudiante,
    COUNT(DISTINCT e.id_estudiante) AS total_estudiantes,
    ROUND(AVG(i.calificacion_final), 2) AS promedio_general,
    MAX(i.calificacion_final) AS mejor_nota,
    MIN(i.calificacion_final) AS peor_nota,
    SUM(i.calificacion_final) AS suma_total_notas
FROM estudiante e
JOIN inscripcion i ON e.id_estudiante = i.id_estudiante
GROUP BY e.carrera_estudiante;

-- 11. Creacion vista
CREATE VIEW vista_historial_academico AS
SELECT 
    e.nombre_completo AS nombre_estudiante,
    c.nombre_curso,
    d.nombre_completo AS nombre_docente,
    c.semestre,
    i.calificacion_final
FROM inscripcion i
JOIN estudiante e 
    ON i.id_estudiante = e.id_estudiante
JOIN curso c 
    ON i.id_curso = c.id_curso
JOIN docente d 
    ON c.id_docente = d.id_docente;
